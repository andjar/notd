    # Create multiple notes in one request
curl -X POST http://localhost/api/v1/notes \
  -H "Content-Type: application/json" \
  -d '{
    "action": "batch",
    "operations": [
      {
        "type": "create",
        "payload": {
          "page_id": 1,
          "content": "First note {tags::important}",
          "order_index": 1
        }
      },
      {
        "type": "create",
        "payload": {
          "page_id": 1,
          "content": "Second note",
          "order_index": 2
        }
      }
    ]
  }'
